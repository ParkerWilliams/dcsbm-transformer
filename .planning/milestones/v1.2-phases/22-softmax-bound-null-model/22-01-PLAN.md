```python
def compute_theoretical_bound(
    qkt_fro_norm: float, v_spectral_norm: float, wo_spectral_norm: float,
    d_k: int, epsilon: float,
) -> float:
    """Bound = epsilon * qkt_fro_norm * v_spectral_norm * wo_spectral_norm / 2.0"""

def inject_perturbation(
    qkt_scaled: torch.Tensor, values: torch.Tensor, wo_weight: torch.Tensor,
    perturbation: torch.Tensor, causal_mask: torch.Tensor,
) -> torch.Tensor:
    """Inject perturbation into scaled QK^T and recompute AVWo. Returns [T, D]."""

def compute_spectral_change(
    avwo_original: torch.Tensor, avwo_perturbed: torch.Tensor,
) -> float:
    """||sigma(perturbed) - sigma(original)||_2 (L2 norm of SV difference vector)."""

def generate_adversarial_direction(
    qkt_scaled: torch.Tensor, causal_mask: torch.Tensor,
) -> torch.Tensor:
    """Adversarial direction from SVD of causally-masked QK^T. Zero upper triangle, unit Frobenius norm."""
```

From docs/softmax_bound.tex, Theorem 6.1 (the main bound):
LaTeX: ||Delta(AVW_O)||_F <= epsilon * ||QK^T||_F * ||V||_2 * ||W_O||_2 / (2*sqrt(d_k))
Code: epsilon * qkt_fro_norm * v_spectral_norm * wo_spectral_norm / 2.0
Note: Code uses SCALED QK^T (already divided by sqrt(d_k)), so sqrt(d_k) cancels.
